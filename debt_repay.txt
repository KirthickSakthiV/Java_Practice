import java.util.*; class Main{   public static void main(String args[]){     Scanner m=new Scanner (System.in);     float a,r;     int y;     a=m.nextFloat();     r=m.nextFloat();     y=m.nextInt();     float i,amt,d,s;     i=(a*r*y)/100;     amt=a+i;     d=(i*2)/100;     s=amt-d;     System.out.printf("%.2f\n",i);     System.out.printf("%.2f\n",amt);     System.out.printf("%.2f\n",d);     System.out.printf("%.2f\n",s);   } }
